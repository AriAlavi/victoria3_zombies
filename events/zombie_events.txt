namespace = zombie

zombie.1 = {
	type = country_event
	placement = ROOT
	hidden = yes

	trigger = {
		exists = c:ZOM
		this = c:ZOM
	}

	immediate = {
		every_country = {
			limit = {
				is_adjacent = ROOT
				NOT = { has_war_with = ROOT }
				NOT = { country_has_primary_culture = c:ZOM }
				total_population > 0
				exists = yes
			}
			change_relations = {
				country = ROOT
				value = -400
			}
			save_scope_as = target_country
			root = {
				create_diplomatic_play = {
					name = zombie_invasion
					war = yes
					type = dp_humiliation
					target_state = scope:target_country.capital
					
					add_war_goal = {
						holder = scope:target_country
						type = humiliation
						target_state = root.capital
					}
					
				}	
			}

		}
	}
}

zombie.2 = {
	hidden = yes
	placement = ROOT	
	trigger = {
		exists = c:ZOM
	}
	
	immediate = {
		every_diplomatic_play = {
			limit = {
				OR = {
					target = c:ZOM
					initiator = c:ZOM
				}
			}
			war = {
				if = {
					limit = { 
						attacker_warleader = {
							total_population > 10000
						} 
					}
					add_war_war_support = {
						target = attacker_warleader
						value = 1000
					}
					add_war_exhaustion = {
						target = attacker_warleader
						value = -1000
					}
				}
				else = {
					prev = {
						end_play = yes
					}
				}
				if = {
					limit = { 
						defender_warleader = {
							total_population > 10000
						} 
					}
					add_war_war_support = {
						target = defender_warleader
						value = 100
					}
					add_war_exhaustion = {
						target = defender_warleader
						value = -10
					}
				}else = {
					prev = {
						end_play = yes
					}
				}

			}	
		}
	}
}
