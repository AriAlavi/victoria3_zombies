namespace = zombie

zombie.1 = {
	type = country_event
	placement = ROOT
	hidden = yes

	trigger = {
		exists = c:ZOM
		exists = yes
		NOT = {
			country_has_primary_culture = cu:zombie
		}
	}

	immediate = {
		every_country = {
			limit = {
				is_adjacent = ROOT
				country_has_primary_culture = cu:zombie
				NOT = { has_war_with = ROOT }
			}
			save_scope_as = zombie_country
		}
		if = {
			limit = {
				exists = scope:zombie_country
			}
			if = {
				limit = {
					is_subject = yes
					NOT = { is_subject_type = subject_type_customs_union }
				}
				make_independent = yes
			}
			scope:zombie_country = {
				create_diplomatic_play = {
					name = zombie_invasion
					war = yes
					type = dp_humiliation
					target_state =root.capital
					
					add_war_goal = {
						holder = root
						type = humiliation
						target_state = scope:zombie_country.capital
					}
					
				}		
			}

		
		}
	}
}


zombie.2 = {
	hidden = yes
	placement = ROOT	
	trigger = {
		exists = c:ZOM
	}
	
	immediate = {
		every_diplomatic_play = {
			limit = {
				OR = {
					target = {
						country_has_primary_culture = cu:zombie
					}
					initiator = {
						country_has_primary_culture = cu:zombie
					}
				}
			}
			war = {
				if = {
					limit = { 
						attacker_warleader = {
							total_population > 10000
						} 
					}
					add_war_war_support = {
						target = attacker_warleader
						value = 1000
					}
					add_war_exhaustion = {
						target = attacker_warleader
						value = -1000
					}
				}
				else = {
					prev = {
						end_play = yes
					}
				}
				if = {
					limit = { 
						defender_warleader = {
							total_population > 10000
						} 
					}
					add_war_war_support = {
						target = defender_warleader
						value = 100
					}
					add_war_exhaustion = {
						target = defender_warleader
						value = -10
					}
				}else = {
					prev = {
						end_play = yes
					}
				}

			}	
		}
	}
}

zombie.7 = {
	type = country_event
	placement = ROOT
	hidden = yes
	trigger = {
		exists = c:ZOM
		this = {
			NOT = { is_country_type = decentralized }
		}
	}
	immediate = {
		every_scope_state = {
			limit = {
				exists = yes
				OR = {
					AND = {
						OR = {
							state_population < 10000
							state_population > 50000000
						}
						owner = {
							country_has_primary_culture = cu:zombie
						}
					}
					state_population > 500000000
				}

				
			}
			create_country = {
				state = this
				tag = BIE
				on_created = {
					if = {
						limit = {
							PREV = {
								state_population > 500000000
							}
						}
						PREV = {
							add_modifier = {
								name = zombie_desolation
							}
						}

					}
					add_modifier = {
						name = zombie_nation_modifier
					}
				}
			}

		}
	}
}

	type = country_event
	placement = root
	hidden = yes
	
	trigger = {
		total_population > 1000000

		NOT = { is_country_type = decentralized }
		num_states > 2
		is_player = no
		year < 1838
	}


	immediate = {
		random_list = {
			30 = {
				if = {
					limit = {
						year > 1837

					set_global_variable  = too_many_zombies_spawned
